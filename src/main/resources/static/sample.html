`; totalProfit += journal.profit; // Append elements flexContainer.appendChild(shoeHeading); flexContainer.appendChild(handProfitContainer); journalDiv.appendChild(flexContainer); document.getElementById('journalContainer').appendChild(journalDiv); // Add glowing effect to the first valid journal if (index === 0) { journalDiv.classList.add('glow'); } playedShoe++; });
